#Iniciar Kafka y zookeeper
docker compose up -d zookeeper kafka

# Esperar 45 segundos
timeout /t 45

# Verificar que Kafka esté corriendo
docker ps

#Topics
docker compose run kafka-init

# Iniciar las aplicaciones
docker compose up -d central driver cp_engine cp_monitor

# Esperar 30 segundos para que se conecten a Kafka
timeout /t 30

# Verificar que todos estén corriendo
docker ps




PARAR:
# Parar todo
docker compose down --remove-orphans

# Verificar que esté limpio
docker ps -a
(Si quedan cosas poner: docker container prune)
